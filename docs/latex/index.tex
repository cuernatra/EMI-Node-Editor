\begin{DoxyEnumerate}

\begin{DoxyCode}{0}

\item Go to build directory:\+


\begin{DoxyCode}{0}
\DoxyCodeLine{cd\ build}

\begin{DoxyCode}{0}
\DoxyCodeLine{cmake\ -\/DBUILD\_SHARED\_LIBS=OFF\ ..}

\end{DoxyCode}

\item Build project:\+


\begin{DoxyCode}{0}
\DoxyCodeLine{cmake\ -\/-\/build\ .}

\end{DoxyCode}

\end{DoxyEnumerate}
